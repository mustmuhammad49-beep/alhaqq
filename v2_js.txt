<script>
/* ═══════════════════════════════════════════════════════════════
   AL-HAQQ v2.0 — MASTERY UPDATE
   Mastery ranks · bookmarks · daily myth · streaks · drill mode ·
   teaser paywall · copy comment reply · reading size · polish
   Pure additive layer: wraps the existing render pipeline,
   injects all new UI at runtime, stores everything in localStorage.
   ═══════════════════════════════════════════════════════════════ */
(function () {
  'use strict';

  /* ── storage ── */
  var S = {
    get: function (k, fb) {
      try { var v = JSON.parse(localStorage.getItem(k)); return (v === null || v === undefined) ? fb : v; }
      catch (e) { return fb; }
    },
    set: function (k, v) { try { localStorage.setItem(k, JSON.stringify(v)); } catch (e) {} }
  };

  var mastered = new Set(S.get('ahq_mastered', []));
  var savedSet = new Set(S.get('ahq_saved', []));
  var reviewPile = new Set(S.get('ahq_review', []));
  var savedOnly = false;

  function persist() {
    S.set('ahq_mastered', Array.from(mastered));
    S.set('ahq_saved', Array.from(savedSet));
    S.set('ahq_review', Array.from(reviewPile));
  }

  function esc(s) {
    return String(s).replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;').replace(/"/g, '&quot;');
  }

  /* ── entry pools ── */
  function nonEmpty() {
    return entries.filter(function (e) { return e.claim && e.rebuttal && !e.interactive; });
  }
  function accessible() {
    return nonEmpty().filter(function (e) { return !e.locked || hasAccess(); });
  }

  /* ── ranks ── */
  var RANKS = [[18, 'Guardian of Truth'], [12, 'Debater'], [7, 'Defender'], [3, 'Student of the Ayat'], [0, 'Seeker']];
  function rankFor(n, total) {
    if (total > 0 && n >= total) return 'Hafiz al-Haqq';
    for (var i = 0; i < RANKS.length; i++) { if (n >= RANKS[i][0]) return RANKS[i][1]; }
    return 'Seeker';
  }

  /* ── streak ── */
  function dayKey(d) { return d.toISOString().slice(0, 10); }
  function tickStreak() {
    var today = new Date();
    var tk = dayKey(today);
    var last = S.get('ahq_last_day', null);
    var streak = S.get('ahq_streak', 0);
    if (last !== tk) {
      var y = new Date(today); y.setDate(y.getDate() - 1);
      streak = (last === dayKey(y)) ? streak + 1 : 1;
      S.set('ahq_streak', streak);
      S.set('ahq_last_day', tk);
    }
    return streak;
  }

  /* ── toast ── */
  var toastEl = null, toastTimer = null;
  function toast(msg) {
    if (!toastEl) { toastEl = document.createElement('div'); toastEl.className = 'v2-toast'; document.body.appendChild(toastEl); }
    toastEl.textContent = msg;
    toastEl.classList.add('visible');
    clearTimeout(toastTimer);
    toastTimer = setTimeout(function () { toastEl.classList.remove('visible'); }, 2200);
  }

  function flashBtn(btn, label) {
    if (!btn) return;
    var old = btn.innerHTML;
    btn.innerHTML = label;
    setTimeout(function () { btn.innerHTML = old; }, 1700);
  }

  /* ── unlock card (teaser paywall) ── */
  function unlockCardHTML() {
    var total = nonEmpty().length;
    return '<div class="v2-unlock-card">' +
      '<div class="v2-unlock-head">🔒 The full rebuttal is for Founding Members</div>' +
      '<ul class="v2-unlock-list">' +
      '<li>All ' + total + ' rebuttals, fully sourced</li>' +
      '<li>Drill Mode on every entry</li>' +
      '<li>Jibril, the research AI</li>' +
      '<li>New myths dismantled weekly</li>' +
      '</ul>' +
      '<button class="v2-unlock-btn">Unlock everything · $2.99/mo</button>' +
      '<div class="v2-unlock-fine">Price locked for life · cancel anytime · rises to $6.99 after 1,000 members</div>' +
      '</div>';
  }

  /* ── decorate rendered cards ── */
  function decorate() {
    document.querySelectorAll('.entry').forEach(function (el) {
      var id = parseInt(el.dataset.id, 10);
      var entry = entries.find(function (e) { return e.id === id; });
      if (!entry || entry.interactive) return;

      var header = el.querySelector('.entry-header');
      if (header && !header.querySelector('.v2-star')) {
        var star = document.createElement('button');
        star.className = 'v2-star' + (savedSet.has(id) ? ' on' : '');
        star.textContent = savedSet.has(id) ? '★' : '☆';
        star.title = 'Save to your arsenal';
        star.setAttribute('aria-label', 'Save this entry');
        star.addEventListener('click', function (ev) { ev.stopPropagation(); toggleSave(id, star); });
        var tog = header.querySelector('.entry-toggle');
        header.insertBefore(star, tog || null);
      }

      var meta = el.querySelector('.entry-meta');
      if (meta) {
        var hasTag = meta.querySelector('.v2-mastered-tag');
        if (mastered.has(id) && !hasTag) {
          var b = document.createElement('span');
          b.className = 'difficulty v2-mastered-tag';
          b.textContent = '✓ Mastered';
          meta.appendChild(b);
        } else if (!mastered.has(id) && hasTag) {
          hasTag.remove();
        }
      }

      if (el.classList.contains('locked')) {
        if (entry.claim && entry.rebuttal && !el.querySelector('.v2-teaser')) {
          var overlay = el.querySelector('.lock-overlay');
          if (overlay) {
            var plain = entry.rebuttal.replace(/<[^>]+>/g, ' ').replace(/\s+/g, ' ').trim();
            overlay.classList.add('v2-unlock');
            overlay.innerHTML = '<div class="v2-teaser">' + esc(plain.slice(0, 300)) + '…</div>' + unlockCardHTML();
          }
        }
      } else {
        var actions = el.querySelector('.entry-actions');
        if (actions && !actions.querySelector('.v2-master-btn')) {
          var mb = document.createElement('button');
          mb.className = 'action-btn v2-master-btn' + (mastered.has(id) ? ' on' : '');
          mb.innerHTML = mastered.has(id) ? '✓ Mastered' : '◇ Mark as mastered';
          mb.addEventListener('click', function (ev) { ev.stopPropagation(); toggleMaster(id, mb, el); });
          actions.appendChild(mb);

          var cr = document.createElement('button');
          cr.className = 'action-btn';
          cr.innerHTML = '💬 Copy comment reply';
          cr.addEventListener('click', function (ev) { ev.stopPropagation(); copyReply(id, cr); });
          actions.appendChild(cr);
        }
      }
    });
  }

  function toggleSave(id, starBtn) {
    if (savedSet.has(id)) {
      savedSet.delete(id);
      if (starBtn) { starBtn.classList.remove('on'); starBtn.textContent = '☆'; }
      toast('Removed from your arsenal');
    } else {
      savedSet.add(id);
      if (starBtn) { starBtn.classList.add('on'); starBtn.textContent = '★'; }
      toast('★ Saved to your arsenal');
    }
    persist();
    if (savedOnly) window.filterEntries();
  }

  function toggleMaster(id, btn, card) {
    if (mastered.has(id)) {
      mastered.delete(id);
      if (btn) { btn.classList.remove('on'); btn.innerHTML = '◇ Mark as mastered'; }
      toast('Moved back to training');
    } else {
      mastered.add(id);
      reviewPile.delete(id);
      if (btn) { btn.classList.add('on'); btn.innerHTML = '✓ Mastered'; }
      var total = nonEmpty().length;
      var n = countMastered();
      toast(n >= total ? '🏆 Hafiz al-Haqq. Every myth mastered.' : '✓ Locked into memory · ' + n + ' of ' + total);
    }
    persist();
    if (card) {
      var meta = card.querySelector('.entry-meta');
      if (meta) {
        var tag = meta.querySelector('.v2-mastered-tag');
        if (mastered.has(id) && !tag) {
          var b = document.createElement('span');
          b.className = 'difficulty v2-mastered-tag';
          b.textContent = '✓ Mastered';
          meta.appendChild(b);
        } else if (!mastered.has(id) && tag) {
          tag.remove();
        }
      }
    }
    updateStrip();
  }

  function countMastered() {
    var n = 0;
    nonEmpty().forEach(function (e) { if (mastered.has(e.id)) n++; });
    return n;
  }

  /* ── copy comment reply ── */
  function copyReply(id, btn) {
    var e = entries.find(function (x) { return x.id === id; });
    if (!e || !e.rebuttal) return;
    var plain = e.rebuttal
      .replace(/<br\s*\/?>/gi, '\n')
      .replace(/<[^>]+>/g, '')
      .replace(/&nbsp;/g, ' ').replace(/&amp;/g, '&').replace(/&quot;/g, '"').replace(/&#39;/g, "'")
      .replace(/[ \t]+/g, ' ')
      .replace(/\n{3,}/g, '\n\n')
      .trim();
    if (plain.length > 900) {
      var cut = plain.slice(0, 900);
      var dot = Math.max(cut.lastIndexOf('. '), cut.lastIndexOf('? '), cut.lastIndexOf('! '));
      plain = dot > 300 ? cut.slice(0, dot + 1) : cut + '…';
    }
    var txt = plain +
      '\n\nFull breakdown with sources: https://alhaqq.it.com/?myth=' + id +
      '\n\nPeople can reject this. Rejecting it does not make it incorrect.';
    if (navigator.clipboard && navigator.clipboard.writeText) {
      navigator.clipboard.writeText(txt).then(function () {
        flashBtn(btn, '✓ Reply copied'); toast('Reply copied · paste it under their comment');
      }).catch(function () { toast('Copy failed · try the share button'); });
    } else {
      try {
        var ta = document.createElement('textarea');
        ta.value = txt; document.body.appendChild(ta); ta.select();
        document.execCommand('copy'); document.body.removeChild(ta);
        flashBtn(btn, '✓ Reply copied'); toast('Reply copied · paste it under their comment');
      } catch (err) { toast('Copy failed · try the share button'); }
    }
  }

  /* ── progress strip ── */
  function buildStrip() {
    var stats = document.querySelector('.stats-bar');
    if (!stats || document.getElementById('v2Strip')) return;
    var strip = document.createElement('div');
    strip.className = 'v2-strip';
    strip.id = 'v2Strip';
    strip.innerHTML =
      '<div class="v2-strip-box">' +
        '<div class="v2-ring" id="v2Ring"><span id="v2RingPct">0%</span></div>' +
        '<div class="v2-strip-info">' +
          '<div class="v2-rank">Rank: <em id="v2Rank">Seeker</em></div>' +
          '<div class="v2-strip-sub" id="v2RankSub">0 of 0 myths mastered</div>' +
        '</div>' +
        '<div class="v2-streak"><strong id="v2Streak">1</strong> DAY STREAK 🔥</div>' +
        '<div class="v2-strip-actions">' +
          '<button class="v2-drillbtn" id="v2DrillOpen">⚔ Drill Mode</button>' +
          '<div class="v2-font" title="Reading size">' +
            '<button id="v2FontDown" aria-label="Smaller text">A−</button>' +
            '<button id="v2FontUp" aria-label="Larger text">A+</button>' +
          '</div>' +
        '</div>' +
      '</div>';
    stats.parentNode.insertBefore(strip, stats.nextSibling);
    document.getElementById('v2DrillOpen').addEventListener('click', openDrill);
    document.getElementById('v2FontDown').addEventListener('click', function () { stepFont(-1); });
    document.getElementById('v2FontUp').addEventListener('click', function () { stepFont(1); });
  }

  function updateStrip() {
    var ring = document.getElementById('v2Ring');
    if (!ring) return;
    var total = nonEmpty().length;
    var n = countMastered();
    var pct = total ? Math.round((n / total) * 100) : 0;
    ring.style.background = 'conic-gradient(var(--gold) ' + pct + '%, rgba(201,168,76,0.12) 0)';
    document.getElementById('v2RingPct').textContent = pct + '%';
    document.getElementById('v2Rank').textContent = rankFor(n, total);
    document.getElementById('v2RankSub').textContent = n + ' of ' + total + ' myths mastered';
    document.getElementById('v2Streak').textContent = S.get('ahq_streak', 1);
  }

  /* ── reading size ── */
  var FONT_STEPS = ['md', 'lg', 'xl'];
  function applyFont(step) {
    if (step === 'md') document.body.removeAttribute('data-reading');
    else document.body.setAttribute('data-reading', step);
    S.set('ahq_fontsize', step);
  }
  function stepFont(dir) {
    var cur = S.get('ahq_fontsize', 'md');
    var i = Math.min(FONT_STEPS.length - 1, Math.max(0, FONT_STEPS.indexOf(cur) + dir));
    applyFont(FONT_STEPS[i]);
    toast('Reading size: ' + FONT_STEPS[i].toUpperCase());
  }

  /* ── daily myth ── */
  function buildDaily() {
    var pool = nonEmpty();
    if (!pool.length || document.querySelector('.v2-daily')) return;
    var idx = Math.floor(Date.now() / 86400000) % pool.length;
    var e = pool[idx];
    var labels = { christian: 'Christian Claim', sunni: 'Sunni Challenge', secular: 'Secular Critique', orientalist: 'Orientalist' };
    var dateStr = new Date().toLocaleDateString(undefined, { month: 'long', day: 'numeric' });
    var lockedNote = (e.locked && !hasAccess()) ? ' · 🔒 Founding Members' : '';
    var host = document.querySelector('.founding-banner') || document.getElementById('database');
    var d = document.createElement('div');
    d.className = 'v2-daily';
    d.innerHTML =
      '<div class="v2-daily-box">' +
        '<div style="flex:1;min-width:220px;">' +
          '<div class="v2-daily-eyebrow">◈ Today\'s Myth · ' + dateStr + '</div>' +
          '<div class="v2-daily-title">' + esc(e.myth) + '</div>' +
          '<div class="v2-daily-sub">' + (labels[e.source] || '') + ' · ' + esc(e.difficulty) + lockedNote + '</div>' +
        '</div>' +
        '<button class="v2-daily-btn">Read the takedown →</button>' +
      '</div>';
    host.parentNode.insertBefore(d, host);
    d.querySelector('.v2-daily-btn').addEventListener('click', function () { jumpToEntry(e.id); });
  }

  function resetFilters() {
    savedOnly = false;
    currentFilter = 'all';
    currentDifficulty = 'all';
    var si = document.getElementById('searchInput');
    if (si) si.value = '';
    document.querySelectorAll('.controls .filters').forEach(function (row) {
      row.querySelectorAll('.filter-btn').forEach(function (b) { b.classList.remove('active'); });
      var first = row.querySelector('.filter-btn');
      if (first) first.classList.add('active');
    });
    var sc = document.getElementById('v2SavedChip');
    if (sc) sc.classList.remove('active');
  }

  function jumpToEntry(id) {
    var el = document.querySelector('.entry[data-id="' + id + '"]');
    if (!el) { resetFilters(); window.filterEntries(); el = document.querySelector('.entry[data-id="' + id + '"]'); }
    if (!el) return;
    el.scrollIntoView({ behavior: 'smooth', block: 'center' });
    if (!el.classList.contains('locked')) {
      setTimeout(function () { if (!el.classList.contains('open')) toggleEntry(id); }, 380);
    } else {
      el.classList.add('v2-pulse');
      setTimeout(function () { el.classList.remove('v2-pulse'); }, 1600);
    }
  }

  /* ── saved chip ── */
  function buildSavedChip() {
    if (document.getElementById('v2SavedChip')) return;
    var rows = document.querySelectorAll('.controls .filters');
    var row = rows[rows.length - 1];
    if (!row) return;
    var chip = document.createElement('button');
    chip.className = 'filter-btn';
    chip.id = 'v2SavedChip';
    chip.textContent = '★ Saved';
    chip.addEventListener('click', function () {
      savedOnly = !savedOnly;
      chip.classList.toggle('active', savedOnly);
      if (savedOnly && savedSet.size === 0) toast('Nothing saved yet · tap ☆ on any entry');
      window.filterEntries();
    });
    row.appendChild(chip);
  }

  /* ── drill mode ── */
  var drillCurrent = null, drillSession = 0;

  function buildDrill() {
    if (document.getElementById('v2Drill')) return;
    var d = document.createElement('div');
    d.className = 'v2-drill';
    d.id = 'v2Drill';
    d.innerHTML =
      '<div class="v2-drill-inner">' +
        '<div class="v2-drill-top">' +
          '<div><div class="v2-drill-label">⚔ Drill Mode · Debate Training</div>' +
          '<div class="v2-drill-count" id="v2DrillCount"></div></div>' +
          '<button class="v2-drill-close" id="v2DrillClose" aria-label="Close drill mode">✕</button>' +
        '</div>' +
        '<div class="v2-drill-card">' +
          '<div class="v2-drill-meta" id="v2DrillMeta"></div>' +
          '<div class="v2-drill-myth" id="v2DrillMyth"></div>' +
          '<div class="v2-drill-claim" id="v2DrillClaim"></div>' +
          '<div class="v2-drill-prompt" id="v2DrillPrompt">Build your rebuttal in your head, out loud, or on paper. Then check it.</div>' +
          '<button class="v2-drill-reveal" id="v2DrillReveal">Reveal the rebuttal</button>' +
          '<div class="v2-drill-answer" id="v2DrillAnswer">' +
            '<div class="v2-drill-rebuttal" id="v2DrillRebuttal"></div>' +
            '<div class="v2-drill-actions">' +
              '<button class="v2-drill-got" id="v2DrillGot">✓ I had it · locked in</button>' +
              '<button class="v2-drill-again" id="v2DrillAgain">↻ Drill this one again</button>' +
            '</div>' +
          '</div>' +
        '</div>' +
      '</div>';
    document.body.appendChild(d);
    document.getElementById('v2DrillClose').addEventListener('click', closeDrill);
    document.getElementById('v2DrillReveal').addEventListener('click', drillReveal);
    document.getElementById('v2DrillGot').addEventListener('click', function () { drillMark(true); });
    document.getElementById('v2DrillAgain').addEventListener('click', function () { drillMark(false); });
    document.addEventListener('keydown', function (e) {
      if (e.key === 'Escape' && d.classList.contains('visible')) closeDrill();
    });
  }

  function weightedPick() {
    var pool = accessible();
    var bag = [];
    pool.forEach(function (e) {
      var w = reviewPile.has(e.id) ? 3 : (mastered.has(e.id) ? 1 : 2);
      for (var i = 0; i < w; i++) bag.push(e);
    });
    var p = bag[Math.floor(Math.random() * bag.length)];
    if (drillCurrent && pool.length > 1) {
      var guard = 0;
      while (p.id === drillCurrent.id && guard++ < 10) p = bag[Math.floor(Math.random() * bag.length)];
    }
    return p;
  }

  function openDrill() {
    if (accessible().length < 1) { toast('No entries available to drill yet'); return; }
    drillSession = 0;
    document.getElementById('v2Drill').classList.add('visible');
    document.body.style.overflow = 'hidden';
    drillNext();
  }

  function drillNext() {
    drillCurrent = weightedPick();
    if (!drillCurrent) { closeDrill(); return; }
    var labels = { christian: 'Christian Claim', sunni: 'Sunni Challenge', secular: 'Secular Critique', orientalist: 'Orientalist' };
    document.getElementById('v2DrillMeta').innerHTML =
      '<span class="tag tag-' + drillCurrent.source + '">' + (labels[drillCurrent.source] || '') + '</span>' +
      '<span class="difficulty" data-diff="' + esc(drillCurrent.difficulty) + '">' + esc(drillCurrent.difficulty) + '</span>' +
      (reviewPile.has(drillCurrent.id) ? '<span class="difficulty">↻ Review</span>' : '');
    document.getElementById('v2DrillMyth').textContent = drillCurrent.myth;
    document.getElementById('v2DrillClaim').textContent = '"' + drillCurrent.claim.replace(/<[^>]+>/g, '') + '"';
    document.getElementById('v2DrillCount').textContent = 'Card ' + (drillSession + 1) + ' this session · ' + accessible().length + ' in your training pool';
    document.getElementById('v2DrillAnswer').classList.remove('visible');
    document.getElementById('v2DrillReveal').style.display = 'block';
    document.getElementById('v2DrillPrompt').style.display = 'block';
    document.querySelector('#v2Drill .v2-drill-inner').scrollTop = 0;
    document.getElementById('v2Drill').scrollTop = 0;
  }

  function drillReveal() {
    var html = drillCurrent.rebuttal;
    if (typeof linkVerseRefs === 'function') { try { html = linkVerseRefs(html); } catch (e) {} }
    if (drillCurrent.verse) {
      html += '<div class="quran-verse" style="margin-top:1.4rem;">' +
        '<div class="quran-arabic">' + drillCurrent.verse.arabic + '</div>' +
        '<div class="quran-translation">"' + drillCurrent.verse.translation + '"</div>' +
        '<div class="quran-ref">— ' + drillCurrent.verse.ref + '</div></div>';
    }
    document.getElementById('v2DrillRebuttal').innerHTML = html;
    document.getElementById('v2DrillAnswer').classList.add('visible');
    document.getElementById('v2DrillReveal').style.display = 'none';
    document.getElementById('v2DrillPrompt').style.display = 'none';
  }

  function drillMark(got) {
    if (drillCurrent) {
      if (got) { mastered.add(drillCurrent.id); reviewPile.delete(drillCurrent.id); }
      else { reviewPile.add(drillCurrent.id); }
      persist();
    }
    drillSession++;
    updateStrip();
    drillNext();
  }

  function closeDrill() {
    document.getElementById('v2Drill').classList.remove('visible');
    document.body.style.overflow = '';
    window.filterEntries();
    if (drillSession > 0) toast('Session done · ' + drillSession + ' card' + (drillSession === 1 ? '' : 's') + ' drilled');
  }

  /* ── scroll progress + back-to-top ── */
  function buildScrollUI() {
    if (document.querySelector('.v2-topbar')) return;
    var bar = document.createElement('div');
    bar.className = 'v2-topbar';
    document.body.appendChild(bar);
    var fab = document.createElement('button');
    fab.className = 'v2-fab';
    fab.innerHTML = '↑';
    fab.setAttribute('aria-label', 'Back to top');
    fab.addEventListener('click', function () { window.scrollTo({ top: 0, behavior: 'smooth' }); });
    document.body.appendChild(fab);
    window.addEventListener('scroll', function () {
      var h = document.documentElement.scrollHeight - window.innerHeight;
      bar.style.width = (h > 0 ? (window.scrollY / h) * 100 : 0) + '%';
      fab.classList.toggle('visible', window.scrollY > 650);
    }, { passive: true });
  }

  /* ── pipeline hooks ── */
  var _origRender = window.renderEntries;
  window.renderEntries = function (filtered) {
    _origRender(filtered);
    decorate();
    updateStrip();
  };

  window.filterEntries = function () {
    var q = document.getElementById('searchInput').value.toLowerCase();
    var filtered = entries.filter(function (e) {
      var matchFilter = currentFilter === 'all' || e.source === currentFilter;
      var matchDifficulty = currentDifficulty === 'all' || (e.difficulty || '').toLowerCase() === currentDifficulty;
      var rebuttalText = (e.rebuttal || '').replace(/<[^>]+>/g, ' ').toLowerCase();
      var matchSearch = !q || e.myth.toLowerCase().includes(q) || rebuttalText.includes(q) || (e.claim && e.claim.toLowerCase().includes(q));
      var matchSaved = !savedOnly || savedSet.has(e.id);
      return matchFilter && matchDifficulty && matchSearch && matchSaved;
    });
    window.renderEntries(filtered);
  };

  /* unlock CTA delegation (cards re-render constantly) */
  document.addEventListener('click', function (e) {
    var b = e.target.closest && e.target.closest('.v2-unlock-btn');
    if (b) { e.stopPropagation(); if (typeof openModal === 'function') openModal(); }
  });

  /* ── init ── */
  function init() {
    tickStreak();
    buildStrip();
    buildDaily();
    buildSavedChip();
    buildDrill();
    buildScrollUI();
    applyFont(S.get('ahq_fontsize', 'md'));
    window.filterEntries();
  }

  if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', init);
  else init();
})();

</script>
